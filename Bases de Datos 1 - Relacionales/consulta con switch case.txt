SELECT COUNTRY_ID ID,
CASE COUNTRY_ID
WHEN 'AU' THEN 'AUSTRALIA'
WHEN 'BR' THEN 'BRAZIL'
WHEN 'CA' THEN 'CANADA'
WHEN 'IA' THEN 'ITALIA'
WHEN 'JP' THEN 'JAPON'
WHEN 'MX' THEN 'MEXICO'
WHEN 'NL' THEN 'PAISES BAJOS'
WHEN 'SG' THEN 'SINGAPORE'
WHEN 'CN' THEN 'CHINA'
WHEN 'UK' THEN 'REINO UNIDO'
ELSE 'NO CONOCE'
END COUNTRY 
FROM LOCATIONS
ORDER BY COUNTRY_ID
/
